```typescript
// assets/core/scripts/save/LocalStorageSaveBackend.ts
// Wraps sys.localStorage for native iOS/Android and non-FB web builds.
// Uses lazy require() for 'cc' so Jest's jest.mock('cc', ...) works correctly in tests.
import { ISaveBackend } from './ISaveBackend';
import { SaveData } from './SaveData';

const SAVE_KEY = 'bloom_harvest_save_v1';

export class LocalStorageSaveBackend implements ISaveBackend {
    async save(data: SaveData): Promise<void> {
        // eslint-disable-next-line @typescript-eslint/no-var-requires
        const { sys } = require('cc');
        sys.localStorage.setItem(SAVE_KEY, JSON.stringify(data));
    }

    async load(): Promise<SaveData | null> {
        // eslint-disable-next-line @typescript-eslint/no-var-requires
        const { sys } = require('cc');
        const raw: string | null = sys.localStorage.getItem(SAVE_KEY);
        if (!raw) return null;
        try {
            return JSON.parse(raw) as SaveData;
        } catch {
            return null;  // Corrupt data — treat as first run
        }
    }
}
```

```typescript
// assets/core/scripts/save/FBInstantSaveBackend.ts
// Wraps FBInstant.player.setDataAsync / getDataAsync for Facebook Instant Games builds.
// FBInstant is auto-injected as a global by Cocos Creator's fb-instant-games build template.
// Note: setDataAsync promise resolving means data is SCHEDULED, not necessarily committed.
// Do NOT use flushDataAsync for routine saves — it is documented as expensive.
import { ISaveBackend } from './ISaveBackend';
import { SaveData } from './SaveData';

declare const FBInstant: any;

const FB_SAVE_KEY = 'saveData';

export class FBInstantSaveBackend implements ISaveBackend {
    async save(data: SaveData): Promise<void> {
        await FBInstant.player.setDataAsync({ [FB_SAVE_KEY]: data });
    }

    async load(): Promise<SaveData | null> {
        const result = await FBInstant.player.getDataAsync([FB_SAVE_KEY]);
        const data = result[FB_SAVE_KEY] as SaveData | undefined;
        return data ?? null;
    }
}
```

```typescript
// assets/core/scripts/save/SaveSystem.ts
// Single entry point for all save/load operations.
// No cc imports — pure TypeScript class, fully testable with Jest.
// Retry policy (per CONTEXT.md): 1 silent retry after 3 seconds. Log in dev only.
// Non-blocking: save() returns void. Internal async work runs in background.
import { ISaveBackend } from './ISaveBackend';
import { SaveData, DEFAULT_SAVE_DATA } from './SaveData';

const RETRY_DELAY_MS = 3000;

function sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
}

export class SaveSystem {
    private static _instance: SaveSystem | undefined;

    static get instance(): SaveSystem {
        if (!SaveSystem._instance) throw new Error('SaveSystem not initialized — call new SaveSystem() first');
        return SaveSystem._instance;
    }

    private backend: ISaveBackend;
    private isDev: boolean;
    private currentData: SaveData;

    constructor(backend: ISaveBackend, isDev: boolean = false) {
        this.backend = backend;
        this.isDev = isDev;
        this.currentData = { ...DEFAULT_SAVE_DATA };
        SaveSystem._instance = this;
    }

    /**
     * Load saved data from backend. Call once during Boot.onLoad().
     * Always resolves — falls back to DEFAULT_SAVE_DATA on any failure.
     */
    async initialize(): Promise<void> {
        try {
            const loaded = await this.backend.load();
            if (loaded !== null) {
                this.currentData = this.migrate(loaded);
            }
            // null = first run → default state already set in constructor
        } catch {
            // Silent fallback to default state (per CONTEXT.md: no error shown to user)
        }
    }

    /**
     * Returns a frozen snapshot of the current in-memory save data.
     * Gameplay reads from this, not from storage.
     * Returns Readonly to signal callers not to mutate the result.
     */
    getData(): Readonly<SaveData> {
        // Shallow copy to prevent callers from mutating internal state
        return Object.freeze({ ...this.currentData });
    }

    /**
     * Merge patch into in-memory state and trigger a background save.
     * Call after level end or after a flower upgrade/unlock.
     * Do NOT call after every frame — only at meaningful state change points.
     */
    update(patch: Partial<SaveData>): void {
        this.currentData = { ...this.currentData, ...patch };
        this.save();
    }

    /**
     * Trigger a background save. Returns void — NEVER await this.
     * Awaiting would block the UI for up to 6+ seconds on double-failure.
     */
    save(): void {
        void this.doSave();  // fire-and-forget
    }

    private async doSave(): Promise<void> {
        try {
            await this.backend.save(this.currentData);
        } catch {
            // First attempt failed — retry once after delay
            await sleep(RETRY_DELAY_MS);
            try {
                await this.backend.save(this.currentData);
            } catch (e) {
                // Retry also failed: log in dev, silent in production (per CONTEXT.md)
                if (this.isDev) {
                    console.warn('[SaveSystem] Save failed after retry:', e);
                }
            }
        }
    }

    /** Handle schema version differences. Extend as versions increment. */
    private migrate(data: SaveData): SaveData {
        if (data.version === DEFAULT_SAVE_DATA.version) return data;
        // Future migrations: v1 → v2, v2 → v3, etc.
        // Merge loaded data over defaults to fill any new fields
        return { ...DEFAULT_SAVE_DATA, ...data, version: DEFAULT_SAVE_DATA.version };
    }
}
```

```typescript
// assets/core/scripts/save/createSaveBackend.ts
// Factory function — selects the correct backend at boot time.
// Called once from Boot.ts after platform initialization.
// Uses sys.isBrowser + FBInstant global detection (same pattern as Phase 1 PlatformDetector).
```
