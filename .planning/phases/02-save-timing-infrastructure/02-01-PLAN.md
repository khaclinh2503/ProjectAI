```typescript
// assets/core/scripts/timing/TimingService.ts
// TECH-03: All bloom window boundaries use wall-clock milliseconds (Date.now()).
// Never use frame counts or dt accumulation. Gameplay code must call TimingService.instance.now()
// instead of Date.now() directly — this keeps the timing authority centralized.

export interface BloomWindow {
    /** Absolute wall-clock timestamp (ms) when bloom state begins. */
    bloomStartMs: number;
    /** Absolute wall-clock timestamp (ms) when bloom state ends. */
    bloomEndMs: number;
}

export class TimingService {
    private static _instance: TimingService | undefined;

    static get instance(): TimingService {
        if (!TimingService._instance) {
            TimingService._instance = new TimingService();
        }
        return TimingService._instance;
    }

    /**
     * Current wall-clock time in milliseconds.
     * All gameplay timing goes through this method — never call Date.now() directly.
     */
    now(): number {
        return Date.now();
    }

    /**
     * Returns true if the current wall-clock time falls inside the bloom window (inclusive).
     */
    isInBloom(window: BloomWindow): boolean {
        const t = this.now();
        return t >= window.bloomStartMs && t <= window.bloomEndMs;
    }

    /**
     * Milliseconds remaining in the bloom window.
     * Returns a negative value if the window has already passed.
     */
    msRemainingInBloom(window: BloomWindow): number {
        return window.bloomEndMs - this.now();
    }

    /**
     * Build a bloom window that begins `delayMs` from now and lasts `durationMs`.
     * Used by the flower lifecycle system (Phase 3) to schedule bloom states.
     */
    createBloomWindow(delayMs: number, durationMs: number): BloomWindow {
        const now = this.now();
        return {
            bloomStartMs: now + delayMs,
            bloomEndMs: now + delayMs + durationMs,
        };
    }
}
```
