```typescript
describe('TimingService', () => {

        // Reset singleton between tests
        (TimingService as any)._instance = undefined;
        service = TimingService.instance;
    });

        const before = Date.now();

        const after = Date.now();
        expect(t).toBeGreaterThanOrEqual(before);
        expect(t).toBeLessThanOrEqual(after);
    });

    it('isInBloom returns true when current time is inside window', () => {
        const now = Date.now();
        const window = { bloomStartMs: now - 100, bloomEndMs: now + 100 };
        expect(service.isInBloom(window)).toBe(true);
    });

    it('isInBloom returns false when window has passed', () => {
        const now = Date.now();
        const window = { bloomStartMs: now - 500, bloomEndMs: now - 100 };
        expect(service.isInBloom(window)).toBe(false);
    });

    it('isInBloom returns false when window has not started', () => {
        const now = Date.now();
        const window = { bloomStartMs: now + 500, bloomEndMs: now + 1000 };
        expect(service.isInBloom(window)).toBe(false);
    });

    it('createBloomWindow produces a window starting in the future', () => {
        const win = service.createBloomWindow(100, 400);
        const now = Date.now();
        // Allow 50ms slack for test execution time
        expect(win.bloomStartMs).toBeGreaterThan(now - 50);
        expect(win.bloomEndMs - win.bloomStartMs).toBe(400);
    });

    it('msRemainingInBloom returns positive value when inside window', () => {
        const now = Date.now();
        const window = { bloomStartMs: now - 100, bloomEndMs: now + 500 };
        const remaining = service.msRemainingInBloom(window);
        expect(remaining).toBeGreaterThan(0);
        expect(remaining).toBeLessThanOrEqual(500);
    });

    it('msRemainingInBloom returns negative value when window has passed', () => {
        const now = Date.now();
```

**tests/save/SaveSystem.test.ts** — covers TECH-05 (load default, load persisted, update + save trigger):
```typescript
import { SaveSystem } from '../../assets/core/scripts/save/SaveSystem';
import { ISaveBackend } from '../../assets/core/scripts/save/ISaveBackend';
import { SaveData, DEFAULT_SAVE_DATA } from '../../assets/core/scripts/save/SaveData';

function makeMockBackend(loadResult: SaveData | null = null, saveShouldFail = false): ISaveBackend {
    return {
        save: jest.fn().mockImplementation(async () => {
            if (saveShouldFail) throw new Error('Save failed');
        }),
        load: jest.fn().mockResolvedValue(loadResult),
    };
}

describe('SaveSystem', () => {
    beforeEach(() => {
        // Reset singleton between tests
        (SaveSystem as any)._instance = undefined;
    });

    it('initialize() loads default state when backend returns null', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        expect(system.getData().highScore).toBe(0);
        expect(system.getData().version).toBe(DEFAULT_SAVE_DATA.version);
        expect(system.getData().tutorialCompleted).toBe(false);
    });

    it('initialize() loads persisted state when backend returns data', async () => {
        const saved: SaveData = { ...DEFAULT_SAVE_DATA, highScore: 9999 };
        const backend = makeMockBackend(saved);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        expect(system.getData().highScore).toBe(9999);
    });

    it('initialize() falls back to default state when backend throws', async () => {
        const backend: ISaveBackend = {
            save: jest.fn(),
            load: jest.fn().mockRejectedValue(new Error('Load failed')),
        };
        const system = new SaveSystem(backend, false);
        await system.initialize();
        expect(system.getData().highScore).toBe(DEFAULT_SAVE_DATA.highScore);
    });

    it('update() merges patch into current data', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        system.update({ highScore: 500 });
        expect(system.getData().highScore).toBe(500);
    });

    it('update() triggers a save (calls backend.save)', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        system.update({ highScore: 500 });
        // Wait for async save to fire
        await new Promise(resolve => setTimeout(resolve, 10));
        expect(backend.save).toHaveBeenCalled();
    });

    it('save() is non-blocking — returns void synchronously', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        // save() must return void (not Promise<void>) so callers cannot await it
        const result = system.save();
        expect(result).toBeUndefined();
    });

    it('getData() returns a readonly snapshot — mutation does not affect internal state', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        const data = system.getData() as SaveData;
        const originalScore = data.highScore;
        // Attempt to mutate returned object
        (data as any).highScore = 99999;
        // Internal state must not change
        expect(system.getData().highScore).toBe(originalScore);
    });

    it('SaveSystem.instance returns the initialized singleton', async () => {
        const backend = makeMockBackend(null);
        const system = new SaveSystem(backend, false);
        await system.initialize();
        expect(SaveSystem.instance).toBe(system);
    });
});
```

**tests/save/LocalStorageSaveBackend.test.ts** — covers TECH-05 (localStorage interactions):
```typescript
import { LocalStorageSaveBackend } from '../../assets/core/scripts/save/LocalStorageSaveBackend';
import { DEFAULT_SAVE_DATA } from '../../assets/core/scripts/save/SaveData';

// Mock sys.localStorage at module level — LocalStorageSaveBackend must NOT import 'cc' directly.
// If it does, this test will fail at import time, which is the intended guard.
const mockStorage: Record<string, string> = {};
const mockLocalStorage = {
    setItem: jest.fn((key: string, value: string) => { mockStorage[key] = value; }),
    getItem: jest.fn((key: string) => mockStorage[key] ?? null),
    removeItem: jest.fn((key: string) => { delete mockStorage[key]; }),
    clear: jest.fn(() => { Object.keys(mockStorage).forEach(k => delete mockStorage[k]); }),
};

// Inject mock before module loads — LocalStorageSaveBackend reads sys at call time
jest.mock('cc', () => ({
    sys: { localStorage: mockLocalStorage },
}));

describe('LocalStorageSaveBackend', () => {
    let backend: LocalStorageSaveBackend;

    beforeEach(() => {
        mockLocalStorage.setItem.mockClear();
        mockLocalStorage.getItem.mockClear();
        Object.keys(mockStorage).forEach(k => delete mockStorage[k]);
        backend = new LocalStorageSaveBackend();
    });

    it('save() calls sys.localStorage.setItem with JSON-serialized data', async () => {
        await backend.save(DEFAULT_SAVE_DATA);
        expect(mockLocalStorage.setItem).toHaveBeenCalledTimes(1);
        const [key, value] = mockLocalStorage.setItem.mock.calls[0];
        expect(key).toBe('bloom_harvest_save_v1');
        const parsed = JSON.parse(value);
        expect(parsed.version).toBe(DEFAULT_SAVE_DATA.version);
        expect(parsed.highScore).toBe(DEFAULT_SAVE_DATA.highScore);
    });

    it('load() returns null when key is missing', async () => {
        const result = await backend.load();
        expect(result).toBeNull();
    });

    it('load() returns parsed SaveData when key exists', async () => {
        await backend.save(DEFAULT_SAVE_DATA);
        const result = await backend.load();
        expect(result).not.toBeNull();
        expect(result!.version).toBe(DEFAULT_SAVE_DATA.version);
        expect(result!.highScore).toBe(DEFAULT_SAVE_DATA.highScore);
    });

    it('load() returns null when stored value is corrupt JSON', async () => {
        mockStorage['bloom_harvest_save_v1'] = 'not-valid-json{{{';
        const result = await backend.load();
        expect(result).toBeNull();
    });
});
```
