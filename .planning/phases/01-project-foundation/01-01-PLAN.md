    **Step 4 — Write assets/boot/scripts/EventBus.ts.**

    ```typescript
    // EventBus.ts — global typed pub/sub, no external dependencies
    // Used by: any system needing decoupled communication
    // Anti-pattern avoided: direct component references across scenes

    type Handler<T = any> = (data: T) => void;

    export class EventBus {
        private static handlers: Map<string, Handler[]> = new Map();

        static on<T>(event: string, handler: Handler<T>): void {
            if (!this.handlers.has(event)) this.handlers.set(event, []);
            this.handlers.get(event)!.push(handler as Handler);
        }

        static off<T>(event: string, handler: Handler<T>): void {
            const list = this.handlers.get(event);
            if (list) {
                const idx = list.indexOf(handler as Handler);
                if (idx >= 0) list.splice(idx, 1);
            }
        }

        static emit<T>(event: string, data?: T): void {
            const list = this.handlers.get(event);
            if (list) [...list].forEach(h => h(data));
        }
    }
    ```

    **Step 6 — Write assets/boot/scripts/Boot.ts.**

    ```typescript
    // Boot.ts — orchestrates platform init and manager persistence
    // Attached to: root node of Boot.scene (first scene loaded)
    // CRITICAL: addPersistRootNode MUST be called before any director.loadScene()

    import { _decorator, Component, director, game } from 'cc';
    const { ccclass } = _decorator;

    @ccclass('Boot')
    export class Boot extends Component {
        onLoad(): void {
            // Persist this node (and all sibling managers) across all scene transitions.
            // Must happen before any loadScene call — scene load destroys non-persistent nodes.
            game.addPersistRootNode(this.node);

            // Phase 1: Platform adapter wiring happens in Plan 01-02.
            // For now, proceed directly to game-assets bundle load.
            this._bootSequence();
        }

        private async _bootSequence(): Promise<void> {
            // Phase 1 stub — no platform adapter yet.
            // Plan 01-02 will insert: await platformAdapter.initialize()
            // and: await platformAdapter.startGame()
            // before director.loadScene.

            // Lazy-load game-assets bundle, then transition to Gameplay scene.
            // assetManager.loadBundle is the only correct API (cc.loader is removed in 3.x).
            const { assetManager } = await import('cc');
            assetManager.loadBundle('game-assets', (err, _bundle) => {
                if (err) {
                    console.error('[Boot] Failed to load game-assets bundle:', err);
                    return;
                }
                director.loadScene('Gameplay');
            });
        }
    }
    ```

    NOTE: The `assetManager.loadBundle('game-assets', ...)` call will fail until the `game-assets/` folder is marked as a bundle in Task 3 (editor step). This is expected — Task 3 configures the bundle.

    **Step 7 — Write scripts/validate-build.sh.**

    Create `scripts/` at the project root (E:/workspace/ProjectAI/scripts/), NOT inside `assets/`.

    ```bash
    #!/usr/bin/env bash
    # validate-build.sh — Phase 1 success criteria checker
    # Usage:
    #   bash scripts/validate-build.sh           (all automated checks)
    #   bash scripts/validate-build.sh --quick   (SC-4 grep only, no build needed)
    #   bash scripts/validate-build.sh --full    (all checks including build artifact)
    #
    # SC-1 (iOS/Android device deploy): MANUAL — cannot be automated without device CI
    # SC-2 (FB runtime in sandbox): MANUAL — requires FB developer portal
    # SC-3 (payload < 5 MB): Automated after build
    # SC-4 (no direct SDK imports): Automated, no build needed

    set -euo pipefail
    PASS=0; FAIL=0; SKIP=0
    QUICK="${1:-}"

    # ── SC-4: No direct FBInstant calls outside adapters/ ─────────────────────
    echo "Checking SC-4: No direct SDK imports outside adapters/..."
    VIOLATIONS=$(grep -r "FBInstant\." assets/ --include="*.ts" 2>/dev/null \
        | grep -v "adapters/" || true)
    if [ -z "$VIOLATIONS" ]; then
        echo "  [PASS] SC-4: No direct FBInstant calls outside adapters/"
        PASS=$((PASS+1))
    else
        echo "  [FAIL] SC-4: Direct SDK imports found outside adapters/:"
        echo "$VIOLATIONS"
        FAIL=$((FAIL+1))
    fi

    if [ "$QUICK" = "--quick" ]; then
        echo ""
        echo "Quick mode: $PASS passed, $FAIL failed, $SKIP skipped"
        [ "$FAIL" -eq 0 ] && exit 0 || exit 1
    fi

    # ── SC-2 + SC-3: Build artifact checks ────────────────────────────────────
    ZIP="build/fb-instant-games/fb-instant-games.zip"
    echo "Checking SC-2: FB Instant Games build artifact exists..."
    if [ -f "$ZIP" ]; then
        echo "  [PASS] SC-2: Build artifact exists at $ZIP"
        PASS=$((PASS+1))
    else
        echo "  [SKIP] SC-2: $ZIP not found — run FB Instant Games build first"
        SKIP=$((SKIP+1))
    fi

    echo "Checking SC-3: Initial payload < 5 MB (5,242,880 bytes)..."
    if [ -f "$ZIP" ]; then
        # stat -c%s is GNU (Linux/Git Bash); stat -f%z is BSD (macOS)
        SIZE=$(stat -c%s "$ZIP" 2>/dev/null || stat -f%z "$ZIP" 2>/dev/null || echo 0)
        if [ "$SIZE" -gt 0 ] && [ "$SIZE" -lt 5242880 ]; then
            echo "  [PASS] SC-3: Payload ${SIZE} bytes < 5,242,880 bytes (5 MB)"
            PASS=$((PASS+1))
        elif [ "$SIZE" -ge 5242880 ]; then
            echo "  [FAIL] SC-3: Payload ${SIZE} bytes >= 5,242,880 bytes (budget exceeded)"
            echo "  Diagnose: unzip -l $ZIP | sort -k1 -rn | head -20"
            FAIL=$((FAIL+1))
        else
            echo "  [SKIP] SC-3: Could not determine file size"
            SKIP=$((SKIP+1))
        fi
    else
        echo "  [SKIP] SC-3: No build artifact to measure"
        SKIP=$((SKIP+1))
    fi

    # ── Summary ───────────────────────────────────────────────────────────────
    echo ""
    echo "Results: $PASS passed, $FAIL failed, $SKIP skipped"
    echo ""
    echo "Manual verifications required before phase gate:"
    echo "  SC-1: Deploy iOS or Android build to device; confirm blank scene renders without errors"
    echo "  SC-2 (runtime): Upload $ZIP to FB App Dashboard > Instant Games > Web Hosting; confirm game starts"
    echo ""
    [ "$FAIL" -eq 0 ] && exit 0 || exit 1
    ```
