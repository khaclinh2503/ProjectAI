    File structure (exact):
    ```typescript
    import { StorageService } from './StorageService';

    export interface LeaderboardEntry {
        name: string;
        score: number;
        timestamp: number;  // Date.now() at save time (D-02)
    }

    export class LeaderboardService {
        private static readonly MAX_ENTRIES = 10;
        private static readonly STORAGE_KEY = 'leaderboard';
        private static readonly NAME_KEY = 'playerName';
        private static readonly MAX_NAME_LENGTH = 12;

        // --- Player Name (PLAYER-01) ---

        static getPlayerName(): string | null {
            return StorageService.get(LeaderboardService.NAME_KEY);
        }

        static setPlayerName(name: string): void {
            StorageService.set(
                LeaderboardService.NAME_KEY,
                name.slice(0, LeaderboardService.MAX_NAME_LENGTH)  // D-10: truncate
            );
        }

        // --- Leaderboard Read (LB-01) ---

        static getEntries(): LeaderboardEntry[] {
            return LeaderboardService._load();
        }

        // --- getRank (LB-02) ---
        // Returns 1-based rank the score would achieve, or null if not qualifying (D-04, D-05)

        static getRank(score: number): number | null {
            const entries = LeaderboardService._load();
            if (!LeaderboardService._wouldQualify(score, entries)) {
                return null;
            }
            // Simulate insertion to find rank position
            const simulated: LeaderboardEntry[] = [
                ...entries,
                { name: '', score, timestamp: Date.now() }
            ];
            simulated.sort((a, b) => b.score - a.score || b.timestamp - a.timestamp);
            // Find the first entry matching our score+timestamp (the simulated one, newest)
            // Since Date.now() is the newest timestamp, on ties our entry sorts first among same-score
            const rank = simulated.findIndex(
                e => e.name === '' && e.score === score
            ) + 1;
            return rank;
        }

        // --- Save Entry (LB-02) ---
        // Returns achieved rank (1-based) or null if score did not qualify (D-08)

        static saveEntry(name: string, score: number): number | null {
            const entries = LeaderboardService._load();
            if (!LeaderboardService._wouldQualify(score, entries)) {
                return null;  // D-08: strictly below 10th, not inserted
            }
            const entry: LeaderboardEntry = {
                name: name.slice(0, LeaderboardService.MAX_NAME_LENGTH),
                score,
                timestamp: Date.now()
            };
            entries.push(entry);
            entries.sort((a, b) => b.score - a.score || b.timestamp - a.timestamp);  // D-06
            const capped = entries.slice(0, LeaderboardService.MAX_ENTRIES);  // D-07
            LeaderboardService._save(capped);
            // Find achieved rank
            const rank = capped.findIndex(
                e => e.name === entry.name
                  && e.score === entry.score
                  && e.timestamp === entry.timestamp
            ) + 1;
            return rank;
        }

        // --- Private helpers ---

        private static _wouldQualify(score: number, entries: LeaderboardEntry[]): boolean {
            if (entries.length < LeaderboardService.MAX_ENTRIES) return true;  // D-05
            // D-08: strictly below last entry does not qualify
            // Equal scores qualify (D-06 tiebreaker: newer timestamp wins)
            return score >= entries[entries.length - 1].score;
        }

        private static _load(): LeaderboardEntry[] {
            const raw = StorageService.get(LeaderboardService.STORAGE_KEY);
            if (!raw) return [];
            try {
                return JSON.parse(raw) as LeaderboardEntry[];
            } catch {
                return [];  // corrupted data graceful fallback
            }
        }

        private static _save(entries: LeaderboardEntry[]): void {
            StorageService.set(
                LeaderboardService.STORAGE_KEY,
                JSON.stringify(entries)
            );
        }
    }
    ```

    IMPORTANT implementation notes:
    - _wouldQualify uses `>=` (not `>`) because equal scores qualify per D-06/D-08: "strictly below" means `<` disqualifies, `>=` qualifies
    - getRank simulates with a dummy name='' entry; findIndex looks for that dummy — this works because no real entry has name=''... BUT this is fragile. Better approach: use timestamp identity since Date.now() in getRank is unique. However, in tests Date.now may be mocked. Safest: after sorting simulated array, find rank by iterating until score is less than target score. The rank is the first position where the score equals or is the target.

    Actually, simplify getRank:
    ```typescript
    static getRank(score: number): number | null {
        const entries = LeaderboardService._load();
        if (!LeaderboardService._wouldQualify(score, entries)) return null;
        // Count how many existing entries have score strictly greater than this score
        // New entry sorts above same-score entries (newer timestamp), so rank = count_of_higher + 1
        const higherCount = entries.filter(e => e.score > score).length;
        return higherCount + 1;

    ```
    This is simpler and correct: entries with equal score sort below the new entry (D-06: newer timestamp higher), so only strictly-higher entries are above it.
